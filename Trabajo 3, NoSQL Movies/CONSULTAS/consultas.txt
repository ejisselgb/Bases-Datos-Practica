/** CONSULTA PARA MOSTRAR LOS ACTORES DE UNA PELICULA Y BUSCAR SI ESTOS HAN TRABAJADO EN OTRAS (2) **/
db.PELICULAS.find({"nombre": "El Irlandes"}).forEach(function(docComparar){
    const actoresAcomp = docComparar.actores
    const tituloComparar = docComparar.nombre
    for(var i = 0; i < actoresAcomp.length;i++){
        var actor =  actoresAcomp[i].nombre
        var imprimir = actor+" trabajÃ³ en: "+tituloComparar
        db.PELICULAS.find().forEach(function(doc){
            var titulo = doc.nombre
            var actores = doc.actores
            if(titulo === tituloComparar){
            }
            else{
                for(var j = 0; j<actores.length; j++){
                    if(actor === actores[j].nombre){
                        imprimir += ", "+titulo
                    }
                }
            }

        })
        print(imprimir) 
    }
})
